    def LoadingGif(self):
        if self.LoadingGifStarted==False:
            if self.loading==True:
                self.loadingLabel.show()
                self.loadgif.start()
                self.LoadingGifStarted=True
        else:
            if self.loading==False:
                self.loadgif.stop()
                self.loadingLabel.hide()
                self.LoadingGifStarted=False
